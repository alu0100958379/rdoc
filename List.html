def popHead()
        if (@size == 0)
                puts "Lista Vacía"
        else
                drop = @head.value
                (@head.next).prev = nil
                @head = @head.next
                @size = @size - 1
                return drop
        end
end

def push(valor)
        node = Node.new(valor,nil,nil)
        if (@size == 0)
                @tail = node
                node.next = nil
        else
                @head.prev = node
                node.next = @head
        end
        @head = node
        node.prev = nil
        @size = @size + 1
end
